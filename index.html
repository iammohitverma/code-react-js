import React, { useState, useEffect } from "react";

const App = () => {

  let [count, setCount] = useState(0);
  let [countTwo, setCountTwo] = useState(0);


  //only run when countTwo update
  useEffect(() => {
    alert("You Clicked " + countTwo + " Times");
  }, [countTwo]);

  

  let incFun = () => {
    setCount(count = count+1);
  }
  let incFunTwo = () => {
    setCountTwo(countTwo = countTwo+1);
  }

  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={incFun}>
        Click me
      </button>
      <p>You clicked {countTwo} times</p>
      <button onClick={incFunTwo}>
        Click me
      </button>
    </div>
  );
}

export default App;

  • Weather Api in React 
    import React, { useState, useEffect } from 'react';

const WeatherApi = () => {

    const [apiData, setapiData] = useState(null);
    const [searchResult, setSearchResult] = useState("Pinjore");

    const inputChange = (event) => {
        setSearchResult(event.target.value);
    }

    useEffect(() => {
        const getData = async () => {       
            const res = await fetch(`https://api.openweathermap.org/data/2.5/weather?q=${searchResult}&units=metric&appid=3f44c9715ae37b2b4db92d9f1cd8d988`);
            const actualData = await res.json();
            setapiData(actualData.main);
        };
        getData();
    }, [searchResult]);

    return(
        <>
            <section>
                <div className='weather-api'>
                    <form onSubmit={(e)=>e.preventDefault()}>
                        <input type="search" placeholder='Search Weather Report' onChange={inputChange} value={searchResult}/>
                        <input type="submit" value="Search"/>
                    </form>
                    {
                        (apiData) ? (
                            <>
                            <div className='data'>
                                <h2>City: {searchResult}</h2>
                                <p>Temperatue: {apiData.temp}</p>
                            </div>
                            </>
                        ) : (<p>No Data Found</p>)
                    }
                </div>
            </section>
        </>
    )
}

export default WeatherApi;

  • Toggle in React 
    import React, {useState} from 'react';
const UseStates = () => {

    const [isActive, setActive] = useState(false);

    const [name, setName] = useState("Mohit");
    
    const changeVal = () => {
        setActive(!isActive);
        !isActive ? setName("Verma") : setName("Mohit");    
    }

    return(
        <>
            <h1>Hello {name}</h1>
            <button className='btn btn-primary' onClick={changeVal}>Click Me</button>
        </>
    )
};

export default UseStates;
